            placeholder: "",
        }

ad_group create

methods POST
path api/v1/superdsp/ad_groups
status old api
controller api/v1/superdsp/ad_groups_controller.rb create
request 
{
    ad_group:{
        tracking_codes: [
            {
                platform_id: 1,
                value: "",
            },
        ],
    },
}
response 
{
    status: success
}
備註 如果 user 沒有設定追蹤碼，不需傳 tracking_codes 陣列（也就是不需要傳任何 tracking_codes hash）。
若只設定兩個，就只傳兩筆 hash。

ad_group update

methods PUT
path api/v1/superdsp/ad_groups/{ad_group_id}
status old api
controller api/v1/superdsp/ad_groups_controller.rb update
request 
{
    ad_group:{
        tracking_codes: [
            {
                platform_id: 1,
                value: "",
            },
            {
                platform_id: 0,
                value: "",
            },
        ],
    },
}
response 
{
    status: success
}
備註 如果有多筆舊資料，就會有多筆 platform_id = 0 的記錄。
新平台的追蹤碼會有對應的 platform_id，舊資料統一以 0 標示。
